  const history = JSON.parse(localStorage.getItem('history') || '[]');

    calcId: calcId,
    timestamp: new Date().toISOString(),
    inputs: inputs,
    result: result
  });
  // Keep last 50
  if (history.length > 50) history.pop();
  localStorage.setItem('history', JSON.stringify(history));
}
